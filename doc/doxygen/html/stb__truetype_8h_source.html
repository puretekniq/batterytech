00730 int stbtt_FindGlyphIndex(const stbtt_fontinfo *info, int unicode_codepoint)
00731 {
00732    stbtt_uint8 *data = info->data;
00733    stbtt_uint32 index_map = info->index_map;
00734 
00735    stbtt_uint16 format = ttUSHORT(data + index_map + 0);
00736    if (format == 0) { // apple byte encoding
00737       stbtt_int32 bytes = ttUSHORT(data + index_map + 2);
00738       if (unicode_codepoint < bytes-6)
00739          return ttBYTE(data + index_map + 6 + unicode_codepoint);
00740       return 0;
00741    } else if (format == 6) {
00742       stbtt_uint32 first = ttUSHORT(data + index_map + 6);
00743       stbtt_uint32 count = ttUSHORT(data + index_map + 8);
00744       if ((stbtt_uint32) unicode_codepoint >= first && (stbtt_uint32) unicode_codepoint < first+count)
00745          return ttUSHORT(data + index_map + 10 + (unicode_codepoint - first)*2);
00746       return 0;
00747    } else if (format == 2) {
00748       STBTT_assert(0); // @TODO: high-byte mapping for japanese/chinese/korean
00749       return 0;
00750    } else if (format == 4) { // standard mapping for windows fonts: binary search collection of ranges
00751       stbtt_uint16 segcount = ttUSHORT(data+index_map+6) >> 1;
00752       stbtt_uint16 searchRange = ttUSHORT(data+index_map+8) >> 1;
00753       stbtt_uint16 entrySelector = ttUSHORT(data+index_map+10);
00754       stbtt_uint16 rangeShift = ttUSHORT(data+index_map+12) >> 1;
00755       stbtt_uint16 item, offset, start, end;
00756 
00757       // do a binary search of the segments
00758       stbtt_uint32 endCount = index_map + 14;
00759       stbtt_uint32 search = endCount;
00760 
00761       if (unicode_codepoint > 0xffff)
00762          return 0;
00763 
00764       // they lie from endCount .. endCount + segCount
00765       // but searchRange is the nearest power of two, so...
00766       if (unicode_codepoint >= ttUSHORT(data + search + rangeShift*2))
00767          search += rangeShift*2;
00768 
00769       // now decrement to bias correctly to find smallest
00770       search -= 2;
00771       while (entrySelector) {
00772          stbtt_uint16 start, end;
00773          searchRange >>= 1;
00774          start = ttUSHORT(data + search + 2 + segcount*2 + 2);
00775          end = ttUSHORT(data + search + 2);
00776          start = ttUSHORT(data + search + searchRange*2 + segcount*2 + 2);
00777          end = ttUSHORT(data + search + searchRange*2);
00778          if (unicode_codepoint > end)
00779             search += searchRange*2;
00780          --entrySelector;
00781       }
00782       search += 2;
00783 
00784       item = (stbtt_uint16) ((search - endCount) >> 1);
00785 
00786       STBTT_assert(unicode_codepoint <= ttUSHORT(data + endCount + 2*item));
00787       start = ttUSHORT(data + index_map + 14 + segcount*2 + 2 + 2*item);
00788       end = ttUSHORT(data + index_map + 14 + 2 + 2*item);
00789       if (unicode_codepoint < start)
00790          return 0;
00791 
00792       offset = ttUSHORT(data + index_map + 14 + segcount*6 + 2 + 2*item);
00793       if (offset == 0)
00794          return unicode_codepoint + ttSHORT(data + index_map + 14 + segcount*4 + 2 + 2*item);
00795 
00796       return ttUSHORT(data + offset + (unicode_codepoint-start)*2 + index_map + 14 + segcount*6 + 2 + 2*item);
00797    } else if (format == 12) {
00798       stbtt_uint16 ngroups = ttUSHORT(data+index_map+6);
00799       stbtt_int32 low,high;
00800        low = 0; high = (stbtt_int32)ngroups;
00801       // Binary search the right group.
00802       while (low <= high) {
00803          stbtt_int32 mid = low + ((high-low) >> 1); // rounds down, so low <= mid < high
00804          stbtt_uint32 start_char = ttULONG(data+index_map+16+mid*12);
00805          stbtt_uint32 end_char = ttULONG(data+index_map+16+mid*12+4);
00806          if ((stbtt_uint32) unicode_codepoint < start_char)
00807             high = mid-1;
00808          else if ((stbtt_uint32) unicode_codepoint > end_char)
00809             low = mid+1;
00810          else {
00811             stbtt_uint32 start_glyph = ttULONG(data+index_map+16+mid*12+8);
00812             return start_glyph + unicode_codepoint-start_char;
00813          }
00814       }
00815       return 0; // not found
00816    }
00817    // @TODO
00818    STBTT_assert(0);
00819    return 0;
00820 }

00885

00894 
00895       n = 1+ttUSHORT(endPtsOfContours + numberOfContours*2-2);
00896

00899       if (vertices == 0)
00900          return 0;
00901 
00902       next_move = 0;
00903       flagcount=0;
00904

00906       // above, shifted to the end of the array so we won't overwrite it when
00907       // we create our final data starting from the front

00910

00961          y     = (stbtt_int16) vertices[off+i].y;

00977             sy = y;
00978          } else {

00983                cy = y;
00984                was_off = 1;
00985             } else {

00988                else

01013          gidx = ttSHORT(comp); comp+=2;

01015          if (flags & 2) { // XY values
01016             if (flags & 1) { // shorts
01017                mtx[4] = ttSHORT(comp); comp+=2;
01018                mtx[5] = ttSHORT(comp); comp+=2;
01019             } else {
01020                mtx[4] = ttCHAR(comp); comp+=1;
01021                mtx[5] = ttCHAR(comp); comp+=1;
01022             }
01023          }
01024          else {
01025             // @TODO handle matching point
01026             STBTT_assert(0);
01027          }
01028          if (flags & (1<<3)) { // WE_HAVE_A_SCALE
01029             mtx[0] = mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
01030             mtx[1] = mtx[2] = 0;
01031          } else if (flags & (1<<6)) { // WE_HAVE_AN_X_AND_YSCALE
01032             mtx[0] = ttSHORT(comp)/16384.0f; comp+=2;
01033             mtx[1] = mtx[2] = 0;
01034             mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;

01036             mtx[0] = ttSHORT(comp)/16384.0f; comp+=2;
01037             mtx[1] = ttSHORT(comp)/16384.0f; comp+=2;
01038             mtx[2] = ttSHORT(comp)/16384.0f; comp+=2;
01039             mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
01040          }
01041          
01042          // Find transformation scales.
01043          m = (float) sqrt(mtx[0]*mtx[0] + mtx[1]*mtx[1]);
01044          n = (float) sqrt(mtx[2]*mtx[2] + mtx[3]*mtx[3]);
01045 
01046          // Get indexed glyph.
01047          comp_num_verts = stbtt_GetGlyphShape(info, gidx, &comp_verts);
01048          if (comp_num_verts > 0) {
01049             // Transform vertices.
01050             for (i = 0; i < comp_num_verts; ++i) {
01051                stbtt_vertex* v = &comp_verts[i];
01052                stbtt_vertex_type x,y;
01053                x=v->x; y=v->y;
01054                v->x = (stbtt_vertex_type)(m * (mtx[0]*x + mtx[2]*y + mtx[4]));
01055                v->y = (stbtt_vertex_type)(n * (mtx[1]*x + mtx[3]*y + mtx[5]));
01056                x=v->cx; y=v->cy;
01057                v->cx = (stbtt_vertex_type)(m * (mtx[0]*x + mtx[2]*y + mtx[4]));
01058                v->cy = (stbtt_vertex_type)(n * (mtx[1]*x + mtx[3]*y + mtx[5]));
01059             }
01060             // Append vertices.
01061             tmp = (stbtt_vertex*)STBTT_malloc((num_vertices+comp_num_verts)*sizeof(stbtt_vertex), info->userdata);
01062             if (!tmp) {
01063                if (vertices) STBTT_free(vertices, info->userdata);
01064                if (comp_verts) STBTT_free(comp_verts, info->userdata);
01065                return 0;
01066             }
01067             if (num_vertices > 0) memcpy(tmp, vertices, num_vertices*sizeof(stbtt_vertex));
01068             memcpy(tmp+num_vertices, comp_verts, comp_num_verts*sizeof(stbtt_vertex));
01069             if (vertices) STBTT_free(vertices, info->userdata);
01070             vertices = tmp;
01071             STBTT_free(comp_verts, info->userdata);
01072             num_vertices += comp_num_verts;
01073          }
01074          // More components ?
01075          more = flags & (1<<5);
01076       }
01077    } else if (numberOfContours < 0) {
01078       // @TODO other compound variations?
01079       STBTT_assert(0);
01080    } else {
01081       // numberOfCounters == 0, do nothing
01082    }
01083 
01084    *pvertices = vertices;
01085    return num_vertices;
01086 }

01181 static stbtt__active_edge *new_active(stbtt__edge *e, int off_x, float start_point, void *userdata)
01182 {
01183    stbtt__active_edge *z = (stbtt__active_edge *) STBTT_malloc(sizeof(*z), userdata); // @TODO: make a pool of these!!!
01184    float dxdy = (e->x1 - e->x0) / (e->y1 - e->y0);
01185    STBTT_assert(e->y0 <= start_point);
01186    if (!z) return z;
01187    // round dx down to avoid going too far
01188    if (dxdy < 0)
01189       z->dx = -STBTT_ifloor(FIX * -dxdy);
01190    else
01191       z->dx = STBTT_ifloor(FIX * dxdy);
01192    z->x = STBTT_ifloor(FIX * (e->x0 + dxdy * (start_point - e->y0)));
01193    z->x -= off_x * FIX;
01194    z->ey = e->y1;
01195    z->next = 0;
01196    z->valid = e->invert ? 1 : -1;
01197    return z;

01200 // note: this routine clips fills that extend off the edges... ideally this
01201 // wouldn't happen, but it could happen if the truetype glyph bounding boxes
01202 // are wrong, or if the user supplies a too-small bitmap
01203 static void stbtt__fill_active_edges(unsigned char *scanline, int len, stbtt__active_edge *e, int max_weight)
01204 {
01205    // non-zero winding fill
01206    int x0=0, w=0;
01207 
01208    while (e) {
01209       if (w == 0) {
01210          // if we're currently at zero, we need to record the edge start point
01211          x0 = e->x; w += e->valid;
01212       } else {
01213          int x1 = e->x; w += e->valid;
01214          // if we went to zero, we need to draw
01215          if (w == 0) {
01216             int i = x0 >> FIXSHIFT;
01217             int j = x1 >> FIXSHIFT;
01218 
01219             if (i < len && j >= 0) {
01220                if (i == j) {
01221                   // x0,x1 are the same pixel, so compute combined coverage
01222                   scanline[i] = scanline[i] + (stbtt_uint8) ((x1 - x0) * max_weight >> FIXSHIFT);
01223                } else {
01224                   if (i >= 0) // add antialiasing for x0
01225                      scanline[i] = scanline[i] + (stbtt_uint8) (((FIX - (x0 & FIXMASK)) * max_weight) >> FIXSHIFT);
01226                   else
01227                      i = -1; // clip
01228 
01229                   if (j < len) // add antialiasing for x1
01230                      scanline[j] = scanline[j] + (stbtt_uint8) (((x1 & FIXMASK) * max_weight) >> FIXSHIFT);
01231                   else
01232                      j = len; // clip
01233 
01234                   for (++i; i < j; ++i) // fill pixels between x0 and x1
01235                      scanline[i] = scanline[i] + (stbtt_uint8) max_weight;
01236                }
01237             }
01238          }
01239       }
01240       
01241       e = e->next;
01242    }
01243 }

01245 static void stbtt__rasterize_sorted_edges(stbtt__bitmap *result, stbtt__edge *e, int n, int vsubsample, int off_x, int off_y, void *userdata)
01246 {
01247    stbtt__active_edge *active = NULL;
01248    int y,j=0;
01249    int max_weight = (255 / vsubsample);  // weight per vertical scanline
01250    int s; // vertical subsample index
01251    unsigned char scanline_data[512], *scanline;
01252 
01253    if (result->w > 512)
01254       scanline = (unsigned char *) STBTT_malloc(result->w, userdata);
01255    else
01256       scanline = scanline_data;
01257 
01258    y = off_y * vsubsample;
01259    e[n].y0 = (off_y + result->h) * (float) vsubsample + 1;
01260 
01261    while (j < result->h) {
01262       STBTT_memset(scanline, 0, result->w);
01263       for (s=0; s < vsubsample; ++s) {
01264          // find center of pixel for this scanline
01265          float scan_y = y + 0.5f;
01266          stbtt__active_edge **step = &active;
01267 
01268          // update all active edges;
01269          // remove all active edges that terminate before the center of this scanline
01270          while (*step) {
01271             stbtt__active_edge * z = *step;
01272             if (z->ey <= scan_y) {
01273                *step = z->next; // delete from list
01274                STBTT_assert(z->valid);
01275                z->valid = 0;
01276                STBTT_free(z, userdata);
01277             } else {
01278                z->x += z->dx; // advance to position for current scanline
01279                step = &((*step)->next); // advance through list
01280             }
01281          }
01282 
01283          // resort the list if needed
01284          for(;;) {
01285             int changed=0;
01286             step = &active;
01287             while (*step && (*step)->next) {
01288                if ((*step)->x > (*step)->next->x) {
01289                   stbtt__active_edge *t = *step;
01290                   stbtt__active_edge *q = t->next;
01291 
01292                   t->next = q->next;
01293                   q->next = t;
01294                   *step = q;
01295                   changed = 1;
01296                }
01297                step = &(*step)->next;
01298             }
01299             if (!changed) break;
01300          }
01301 
01302          // insert all edges that start before the center of this scanline -- omit ones that also end on this scanline
01303          while (e->y0 <= scan_y) {
01304             if (e->y1 > scan_y) {
01305                stbtt__active_edge *z = new_active(e, off_x, scan_y, userdata);
01306                // find insertion point
01307                if (active == NULL)
01308                   active = z;
01309                else if (z->x < active->x) {
01310                   // insert at front
01311                   z->next = active;
01312                   active = z;
01313                } else {
01314                   // find thing to insert AFTER
01315                   stbtt__active_edge *p = active;
01316                   while (p->next && p->next->x < z->x)
01317                      p = p->next;
01318                   // at this point, p->next->x is NOT < z->x
01319                   z->next = p->next;
01320                   p->next = z;
01321                }
01322             }
01323             ++e;
01324          }
01325 
01326          // now process all active edges in XOR fashion
01327          if (active)
01328             stbtt__fill_active_edges(scanline, result->w, active, max_weight);
01329 
01330          ++y;
01331       }
01332       STBTT_memcpy(result->pixels + j * result->stride, scanline, result->w);
01333       ++j;
01334    }
01335 
01336    while (active) {
01337       stbtt__active_edge *z = active;
01338       active = active->next;
01339       STBTT_free(z, userdata);
01340    }
01341 
01342    if (scanline != scanline_data)
01343       STBTT_free(scanline, userdata);
01344 }

01361 static void stbtt__rasterize(stbtt__bitmap *result, stbtt__point *pts, int *wcount, int windings, float scale_x, float scale_y, int off_x, int off_y, int invert, void *userdata)
01362 {
01363    float y_scale_inv = invert ? -scale_y : scale_y;
01364    stbtt__edge *e;
01365    int n,i,j,k,m;
01366    int vsubsample = result->h < 8 ? 15 : 5;
01367    // vsubsample should divide 255 evenly; otherwise we won't reach full opacity
01368 
01369    // now we have to blow out the windings into explicit edge lists
01370    n = 0;
01371    for (i=0; i < windings; ++i)
01372       n += wcount[i];
01373 
01374    e = (stbtt__edge *) STBTT_malloc(sizeof(*e) * (n+1), userdata); // add an extra one as a sentinel
01375    if (e == 0) return;
01376    n = 0;
01377 
01378    m=0;
01379    for (i=0; i < windings; ++i) {
01380       stbtt__point *p = pts + m;
01381       m += wcount[i];
01382       j = wcount[i]-1;
01383       for (k=0; k < wcount[i]; j=k++) {
01384          int a=k,b=j;
01385          // skip the edge if horizontal
01386          if (p[j].y == p[k].y)
01387             continue;
01388          // add edge from j to k to the list
01389          e[n].invert = 0;
01390          if (invert ? p[j].y > p[k].y : p[j].y < p[k].y) {
01391             e[n].invert = 1;
01392             a=j,b=k;
01393          }
01394          e[n].x0 = p[a].x * scale_x;
01395          e[n].y0 = p[a].y * y_scale_inv * vsubsample;
01396          e[n].x1 = p[b].x * scale_x;
01397          e[n].y1 = p[b].y * y_scale_inv * vsubsample;
01398          ++n;
01399       }
01400    }
01401 
01402    // now sort the edges by their highest point (should snap to integer, and then by x)
01403    STBTT_sort(e, n, sizeof(e[0]), stbtt__edge_compare);
01404 
01405    // now, traverse the scanlines and find the intersections on each scanline, use xor winding rule
01406    stbtt__rasterize_sorted_edges(result, e, n, vsubsample, off_x, off_y, userdata);
01407 
01408    STBTT_free(e, userdata);
01409 }

01439 // returns number of contours
01440 stbtt__point *stbtt_FlattenCurves(stbtt_vertex *vertices, int num_verts, float objspace_flatness, int **contour_lengths, int *num_contours, void *userdata)
01441 {
01442    stbtt__point *points=0;
01443    int num_points=0;
01444 
01445    float objspace_flatness_squared = objspace_flatness * objspace_flatness;
01446    int i,n=0,start=0, pass;
01447 
01448    // count how many "moves" there are to get the contour count
01449    for (i=0; i < num_verts; ++i)
01450       if (vertices[i].type == STBTT_vmove)
01451          ++n;
01452 
01453    *num_contours = n;
01454    if (n == 0) return 0;
01455 
01456    *contour_lengths = (int *) STBTT_malloc(sizeof(**contour_lengths) * n, userdata);
01457 
01458    if (*contour_lengths == 0) {
01459       *num_contours = 0;
01460       return 0;
01461    }
01462 
01463    // make two passes through the points so we don't need to realloc
01464    for (pass=0; pass < 2; ++pass) {
01465       float x=0,y=0;
01466       if (pass == 1) {
01467          points = (stbtt__point *) STBTT_malloc(num_points * sizeof(points[0]), userdata);
01468          if (points == NULL) goto error;
01469       }
01470       num_points = 0;
01471       n= -1;
01472       for (i=0; i < num_verts; ++i) {
01473          switch (vertices[i].type) {
01474             case STBTT_vmove:
01475                // start the next contour
01476                if (n >= 0)
01477                   (*contour_lengths)[n] = num_points - start;
01478                ++n;
01479                start = num_points;
01480 
01481                x = vertices[i].x, y = vertices[i].y;
01482                stbtt__add_point(points, num_points++, x,y);
01483                break;
01484             case STBTT_vline:
01485                x = vertices[i].x, y = vertices[i].y;
01486                stbtt__add_point(points, num_points++, x, y);
01487                break;
01488             case STBTT_vcurve:
01489                stbtt__tesselate_curve(points, &num_points, x,y,
01490                                         vertices[i].cx, vertices[i].cy,
01491                                         vertices[i].x,  vertices[i].y,
01492                                         objspace_flatness_squared, 0);
01493                x = vertices[i].x, y = vertices[i].y;
01494                break;
01495          }
01496       }
01497       (*contour_lengths)[n] = num_points - start;
01498    }
01499 
01500    return points;
01501 error:
01502    STBTT_free(points, userdata);
01503    STBTT_free(*contour_lengths, userdata);
01504    *contour_lengths = 0;
01505    *num_contours = 0;

01668 // check if a utf8 string contains a prefix which is the utf16 string; if so return length of matching utf8 string
01669 static stbtt_int32 stbtt__CompareUTF8toUTF16_bigendian_prefix(stbtt_uint8 *s1, stbtt_int32 len1, stbtt_uint8 *s2, stbtt_int32 len2) 
01670 {

01672 
01673    // convert utf16 to utf8 and compare the results while converting
01674    while (len2) {
01675       stbtt_uint16 ch = s2[0]*256 + s2[1];
01676       if (ch < 0x80) {
01677          if (i >= len1) return -1;
01678          if (s1[i++] != ch) return -1;
01679       } else if (ch < 0x800) {
01680          if (i+1 >= len1) return -1;
01681          if (s1[i++] != 0xc0 + (ch >> 6)) return -1;
01682          if (s1[i++] != 0x80 + (ch & 0x3f)) return -1;
01683       } else if (ch >= 0xd800 && ch < 0xdc00) {

01685          stbtt_uint16 ch2 = s2[2]*256 + s2[3];
01686          if (i+3 >= len1) return -1;

01688          if (s1[i++] != 0xf0 + (c >> 18)) return -1;
01689          if (s1[i++] != 0x80 + ((c >> 12) & 0x3f)) return -1;
01690          if (s1[i++] != 0x80 + ((c >>  6) & 0x3f)) return -1;
01691          if (s1[i++] != 0x80 + ((c      ) & 0x3f)) return -1;
01692          s2 += 2; // plus another 2 below
01693          len2 -= 2;
01694       } else if (ch >= 0xdc00 && ch < 0xe000) {
01695          return -1;
01696       } else {
01697          if (i+2 >= len1) return -1;
01698          if (s1[i++] != 0xe0 + (ch >> 12)) return -1;
01699          if (s1[i++] != 0x80 + ((ch >> 6) & 0x3f)) return -1;
01700          if (s1[i++] != 0x80 + ((ch     ) & 0x3f)) return -1;
