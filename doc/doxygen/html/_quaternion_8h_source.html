00542         Quaternion<T> slerp(T r, const Quaternion<T>& q2) const {
00543                 Quaternion<T> ret;
00544                 T cosTheta = w * q2.w + v.x * q2.v.x + v.y * q2.v.y + v.z * q2.v.z;
00545                 T theta = (T) acos(cosTheta);
00546                 if (fabs(theta) < epsilon) {
00547                         ret = *this;
00548                 } else {
00549                         T sinTheta = (T) sqrt(1.0 - cosTheta * cosTheta);
00550                         if (fabs(sinTheta) < epsilon) {
00551                                 ret.w = 0.5 * w + 0.5 * q2.w;
00552                                 ret.v = v.lerp(0.5, q2.v);
00553                         } else {
00554                                 T rA = (T) sin((1.0 - r) * theta) / sinTheta;
00555                                 T rB = (T) sin(r * theta) / sinTheta;
00556 
00557                                 ret.w = w * rA + q2.w * rB;
00558                                 ret.v.x = v.x * rA + q2.v.x * rB;
00559                                 ret.v.y = v.y * rA + q2.v.y * rB;
00560                                 ret.v.z = v.z * rA + q2.v.z * rB;
00561                         }
00562                 }
00563                 return ret;
00564         }
