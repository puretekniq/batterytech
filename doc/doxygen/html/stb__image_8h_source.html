00009 //
00010 // Limitations:
00011 //    - no jpeg progressive support
00012 //    - non-HDR formats support 8-bit samples only (jpeg, png)
00013 //    - no delayed line count (jpeg) -- IJG doesn't support either
00014 //    - no 1-bit BMP
00015 //    - GIF always returns *comp=4
00016 //
00017 // Basic usage (see HDR discussion below):
00018 //    int x,y,n;
00019 //    unsigned char *data = stbi_load(filename, &x, &y, &n, 0);
00020 //    // ... process data if not NULL ...
00021 //    // ... x = width, y = height, n = # 8-bit components per pixel ...
00022 //    // ... replace '0' with '1'..'4' to force that many components per pixel
00023 //    stbi_image_free(data)
00024 //
00025 // Standard parameters:
00026 //    int *x       -- outputs image width in pixels
00027 //    int *y       -- outputs image height in pixels
00028 //    int *comp    -- outputs # of image components in image file
00029 //    int req_comp -- if non-zero, # of image components requested in result
00030 //
00031 // The return value from an image loader is an 'unsigned char *' which points
00032 // to the pixel data. The pixel data consists of *y scanlines of *x pixels,
00033 // with each pixel consisting of N interleaved 8-bit components; the first
00034 // pixel pointed to is top-left-most in the image. There is no padding between
00035 // image scanlines or between pixels, regardless of format. The number of
00036 // components N is 'req_comp' if req_comp is non-zero, or *comp otherwise.
00037 // If req_comp is non-zero, *comp has the number of components that _would_
00038 // have been output otherwise. E.g. if you set req_comp to 4, you will always
00039 // get RGBA output, but you can check *comp to easily see if it's opaque.
00040 //
00041 // An output image with N components has the following components interleaved
00042 // in this order in each pixel:
00043 //
00044 //     N=#comp     components
00045 //       1           grey
00046 //       2           grey, alpha
00047 //       3           red, green, blue
00048 //       4           red, green, blue, alpha
00049 //
00050 // If image loading fails for any reason, the return value will be NULL,
00051 // and *x, *y, *comp will be unchanged. The function stbi_failure_reason()
00052 // can be queried for an extremely brief, end-user unfriendly explanation
00053 // of why the load failed. Define STBI_NO_FAILURE_STRINGS to avoid
00054 // compiling these strings at all, and STBI_FAILURE_USERMSG to get slightly
00055 // more user-friendly ones.
00056 //
00057 // Paletted PNG, BMP, GIF, and PIC images are automatically depalettized.
00058 //
00059 // ===========================================================================
00060 //
00061 // iPhone PNG support:
00062 //
00063 // By default we convert iphone-formatted PNGs back to RGB; nominally they
00064 // would silently load as BGR, except the existing code should have just
00065 // failed on such iPhone PNGs. But you can disable this conversion by
00066 // by calling stbi_convert_iphone_png_to_rgb(0), in which case
00067 // you will always just get the native iphone "format" through.
00068 //
00069 // Call stbi_set_unpremultiply_on_load(1) as well to force a divide per
00070 // pixel to remove any premultiplied alpha *only* if the image file explicitly
00071 // says there's premultiplied data (currently only happens in iPhone images,
00072 // and only if iPhone convert-to-rgb processing is on).
00073 //
00074 // ===========================================================================
00075 //
00076 // HDR image support   (disable by defining STBI_NO_HDR)
00077 //
00078 // stb_image now supports loading HDR images in general, and currently
00079 // the Radiance .HDR file format, although the support is provided
00080 // generically. You can still load any file through the existing interface;
00081 // if you attempt to load an HDR file, it will be automatically remapped to
00082 // LDR, assuming gamma 2.2 and an arbitrary scale factor defaulting to 1;
00083 // both of these constants can be reconfigured through this interface:
00084 //
00085 //     stbi_hdr_to_ldr_gamma(2.2f);
00086 //     stbi_hdr_to_ldr_scale(1.0f);
00087 //
00088 // (note, do not use _inverse_ constants; stbi_image will invert them
00089 // appropriately).
00090 //
00091 // Additionally, there is a new, parallel interface for loading files as
00092 // (linear) floats to preserve the full dynamic range:
00093 //
00094 //    float *data = stbi_loadf(filename, &x, &y, &n, 0);
00095 //
00096 // If you load LDR images through this interface, those images will
00097 // be promoted to floating point values, run through the inverse of
00098 // constants corresponding to the above:
00099 //
00100 //     stbi_ldr_to_hdr_scale(1.0f);
00101 //     stbi_ldr_to_hdr_gamma(2.2f);
00102 //
00103 // Finally, given a filename (or an open file or memory block--see header
00104 // file for details) containing image data, you can query for the "most
00105 // appropriate" interface to use (that is, whether the image is HDR or
00106 // not), using:
00107 //
00108 //     stbi_is_hdr(char *filename);
